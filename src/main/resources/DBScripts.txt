
//Beneficiary Scripts

//Addition of Beneficiary Sequence
CREATE TABLE `banking`.`beneficiary_sequence` (
  `BENEFICIARY_ID` INT NOT NULL AUTO_INCREMENT,
  PRIMARY KEY (`BENEFICIARY_ID`));


//Beneficiary Table
CREATE TABLE `banking`.`beneficiary_info` (
  `TRANSACTION_ID` INT NOT NULL AUTO_INCREMENT,
  `CLIENT_ID` DECIMAL NOT NULL,
  `BENEFICIARY_TYPE` INT NOT NULL,
  `BENEFICIARY_ACCOUNT_NUMBER` VARCHAR(45) NOT NULL,
  `BENEFICIARY_ACCOUNT_TYPE` INT NOT NULL,
  `IFSC_CODE` VARCHAR(45) NOT NULL,
  `BENEFICIARY_NAME` VARCHAR(45) NULL,
  `EMAIL_ID` VARCHAR(100) NULL,
  PRIMARY KEY (`TRANSACTION_ID`));
